"/**

 * @param <T> тип задач

@AllArgsConstructor
@Slf4j
public class ScheduledTaskRunsRepositoryImpl<T extends Enum<T>> implements ScheduledTaskRunsRepository<T> {

    private final DataspaceCorePacketClient dataspacePacketClient;
    private final TaskManagerProperties properties;

    @Override
    public void saveStartTask(ScheduledTaskGetter<T> taskInfo) {
        log.debug("Сохранение информации о запуске задачи {}", taskInfo.getTaskType());
        var packet = Packet.createPacket();
        var param = CreateScheduledTaskRunsParam.create()
                .setTaskId(ScheduledTaskRef.of(taskInfo.getTaskType().name()))
                .setExecutionStatus(ExecutionTaskResultStatus.IN_PROGRESS.name())
                .setTimeStamp(LocalDateTime.now())
                .setResultString("Параметры запуска -" + taskInfo.getParams())
                .setNode(properties.getNodeName());
        packet.scheduledTaskRuns.create(param);
        try {
            dataspacePacketClient.execute(packet);
        } catch (Exception e){
            log.warn("Не удалось сохранить информацию о запуске задачи {}", taskInfo.getTaskType());
        }
    }

    @Override
    public void saveFinishTask(ExecutionResultModel<T> resultInfo) {
        log.debug("Сохранение информации об исполнении задачи {}", resultInfo.getTaskType());
        var packet = Packet.createPacket();
        var param = CreateScheduledTaskRunsParam.create()
                .setTaskId(ScheduledTaskRef.of(resultInfo.getTaskType().name()))
                .setExecutionStatus(resultInfo.getResult().name())
                .setResultString(resultInfo.getResultMessage())
                .setTimeStamp(LocalDateTime.now())
                .setNode(properties.getNodeName())
                .setExecutionDuration(resultInfo.getExecutionDuration());
        packet.scheduledTaskRuns.create(param);
        try {
            dataspacePacketClient.execute(packet);
        } catch (Exception e){
            log.warn("Не удалось сохранить информацию об исполнении задачи {}", resultInfo.getTaskType());
        }
    }
